injector = inject(Injector);

  onToObservableExample() {
    const numbers = signal(0);
    numbers.set(1);
    numbers.set(2);
    numbers.set(3);
    const numbers$ = toObservable(numbers, {
        injector: this.injector,
      })

    numbers.set(4);
    numbers$.subscribe(
      numbers => console.log('numbers$ ', numbers)
    )
    numbers.set(5); 
  }
